### 安装

```js
npm i qzd-utils
//or
yarn add qzd-utils
```

### 使用

```js
// 密码校验使用方法
import { vailPassword } from 'qzd-utils';

vailPassword({ password: 'qzdip.2018', account: 'qzdip' });
```
